 17    def check_trace_conformance(trace: dict, decl_model: DeclareModel, consider_vacuity: bool = False) -> \
 18            List[CheckerResult]:
 19        """
 20        Checks whether the constraints are fulfillment, violation, pendings, activations etc
 21
 22        Parameters
 23        ----------
 24        :param bool consider_vacuity: True means that vacuously satisfied traces are considered as satisfied, violated otherwise
 25        :param d4pyEventLog trace: log
 26        :param DeclareModel decl_model: Process mining model
 27        """
 28
 29        # Set containing all constraints that raised SyntaxError in checker functions
 30        rules = {"vacuous_satisfaction": consider_vacuity}
 31        error_constraint_set = set()
 32        model: DeclareModel = decl_model
 33        trace_results = []
 34        for idx, constraint in enumerate(model.constraints):
 35            constraint_str = model.serialized_constraints[idx]
 36            rules["activation"] = constraint['condition'][0]
 37            if constraint['template'].supports_cardinality:
 38                rules["n"] = constraint['n']
 39            if constraint['template'].is_binary:
 40                rules["correlation"] = constraint['condition'][1]
 41            rules["time"] = constraint['condition'][-1]  # time condition is always at last position
 42            try:
 43                trace_results.append(TemplateConstraintChecker(trace, True, constraint['activities'], rules).get_template(constraint['template'])())
 44            except SyntaxError:
 45                # TODO: use python logger
 46                if constraint_str not in error_constraint_set:
 47                    error_constraint_set.add(constraint_str)
 48                    print('Condition not properly formatted for constraint "' + constraint_str + '".')
 49        return trace_results

 80    def mpChoice(self) -> CheckerResult:
 81        activation_rules = self.declare_parser_utility.parse_data_cond(self.rules["activation"])
 82        time_rule = self.declare_parser_utility.parse_time_cond(self.rules["time"])
 83        a_or_b_occurs = False
 84        for A in self.traces:
 85            if A["concept:name"] == self.activities[0] or A["concept:name"] == self.activities[1]:
 86                locl = {'A': A, 'T': self.traces[0], 'timedelta': timedelta, 'abs': abs, 'float': float}
 87                if eval(activation_rules, glob, locl) and eval(time_rule, glob, locl):
 88                    a_or_b_occurs = True
 89                    break
 90        state = None
 91        if not self.completed and not a_or_b_occurs:
 92            state = TraceState.POSSIBLY_VIOLATED
 93        elif self.completed and not a_or_b_occurs:
 94            state = TraceState.VIOLATED
 95        elif a_or_b_occurs:
 96            state = TraceState.SATISFIED
 97
 98        return CheckerResult(num_fulfillments=None, num_violations=None, num_pendings=None, num_activations=None,

101    def mpExclusiveChoice(self):
102        activation_rules = self.declare_parser_utility.parse_data_cond(self.rules["activation"])
103        time_rule = self.declare_parser_utility.parse_time_cond(self.rules["time"])
104        a_occurs = False
105        b_occurs = False
106        for A in self.traces:
107            locl = {'A': A, 'T': self.traces[0], 'timedelta': timedelta, 'abs': abs, 'float': float}
108            if not a_occurs and A["concept:name"] == self.activities[0]:
109                if eval(activation_rules, glob, locl) and eval(time_rule, glob, locl):
110                    a_occurs = True
111            if not b_occurs and A["concept:name"] == self.activities[1]:
112                if eval(activation_rules, glob, locl) and eval(time_rule, glob, locl):
113                    b_occurs = True
114            if a_occurs and b_occurs:
115                break
116        state = None
117        if not self.completed and (not a_occurs and not b_occurs):
118            state = TraceState.POSSIBLY_VIOLATED
119        elif not self.completed and (a_occurs ^ b_occurs):
120            state = TraceState.POSSIBLY_SATISFIED
121        elif (a_occurs and b_occurs) or (self.completed and (not a_occurs and not b_occurs)):
122            state = TraceState.VIOLATED
123        elif self.completed and (a_occurs ^ b_occurs):
124            state = TraceState.SATISFIED
125
126        return CheckerResult(num_fulfillments=None, num_violations=None, num_pendings=None, num_activations=None,
127                             state=state)
