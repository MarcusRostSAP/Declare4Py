 53        activation_rules = self.declare_parser_utility.parse_data_cond(self.rules["activation"])
 54        time_rule = self.declare_parser_utility.parse_time_cond(self.rules["time"])
 55        a_occurs = False
 56        b_occurs = False
 57        for A in self.traces:
 58            locl = {'A': A, 'T': self.traces[0], 'timedelta': timedelta, 'abs': abs, 'float': float}
 59            if not a_occurs and A["concept:name"] == self.activities[0]:
 60                if eval(activation_rules, glob, locl) and eval(time_rule, glob, locl):
 61                    a_occurs = True
 62            if not b_occurs and A["concept:name"] == self.activities[1]:
 63                if eval(activation_rules, glob, locl) and eval(time_rule, glob, locl):
 64                    b_occurs = True
 65            if a_occurs and b_occurs:
 66                break
 67        state = None
 68        if not self.completed and (not a_occurs and not b_occurs):
 69            state = TraceState.POSSIBLY_VIOLATED
 70        elif not self.completed and (a_occurs ^ b_occurs):
 71            state = TraceState.POSSIBLY_SATISFIED
 72        elif (a_occurs and b_occurs) or (self.completed and (not a_occurs and not b_occurs)):
 73            state = TraceState.VIOLATED
 74        elif self.completed and (a_occurs ^ b_occurs):
 75            state = TraceState.SATISFIED
 76
 77        return CheckerResult(num_fulfillments=None, num_violations=None, num_pendings=None, num_activations=None,
 78                             state=state)
