160    def twogram(self, x: str, y: array, v: float) -> float:
161        """
162        Tranforms the input array y into a 2-gram encoded array:
163        
164        Parameters
165        -------------------
166        x
167            a subsequence from ngram_list
168        y
169            an array from each row in PandasDataframe
170        v
171            a decay factor parameter in n-gram, ranged in [0,1]     
172        ngram_list
173            a list of subsequences for ngram
174            
175        Returns
176        ------------------
177        :rtype: array
178            a calculated score regarding a subsequence x
179        """
180        
181        terms = x.split('|')
182        loc1 = [i for i, e in enumerate(y) if e == terms[0]]
183        layer1 = list()
184        if len(loc1) >0:
185            for i in loc1:      
186                loc2 = [j for j in range(i, len(y)) if y[j] == terms[1]]
187                layer1.append(sum(v**(np.array(loc2) - i)))
188        else:
189            layer1.append(0)
190        
191        return np.sum(layer1)
